function cache(func) {
  let cached = {};
  return function(...args) {
    let key = JSON.stringify(args);
    if(!(key in cached)) cached[key] = func(...args);
    return cached[key];
  }
}
function compose(...funcs) {
  return (n) => {
    for(let i = funcs.length-1;i >= 0;i--){
      n = funcs[i](n);
    }
    return n;
  }
}
